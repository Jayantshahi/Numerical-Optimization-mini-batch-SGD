class Worker:
    def __init__(self, x, y, a, alpha, lam, beta):
        self.x = x #s-by-d is given local feature matrix
        self.y = y #s-by-1 is given local label matrix
        self.s = x.shape[0] #number of local samples
        self.d = x.shape[1] #number of features
        self.w = numpy.zeros((self.d, 1)) #d-by-1 is given model parameter vector
        self.q = a #No of times local gradient will be performed
        self.v = numpy.zeros((self.d, 1)) #d-by1 is given momentum
        self.alpha = alpha #Local learning rate
        self.initial_w = numpy.zeros((self.d, 1)) #To store the w broadcasted by server for finding direction at the end
        self.lam = lam #regularization parameter
        self.beta = beta #momentum parameter 
    #model parameters are updated to the latest
    def set_param(self, w):
        self.w = w
        self.initial_w = w
    
    #compute the local loss
    def loss(self):
        yx = numpy.multiply(self.y, self.x) #s-by-d matrix
        yxw = numpy.dot(yx, self.w) #s-by-1 matrix
        vec1 = numpy.exp(-yxw) #s-by-1 matrix
        vec2 = numpy.log(1 + vec1) #s-by-1 matrix
        return numpy.sum(vec2)/self.s + (self.lam / 2*numpy.sum(self.w * self.w)) #loss function
    
    #Compute the local gradient 
    def gradient(self):
        yx = numpy.multiply(self.y, self.x) # s-by-d matrix
        yxw = numpy.dot(yx, self.w) # s-by-1 matrix
        vec1 = numpy.exp(yxw) # s-by-1 matrix
        vec2 = numpy.divide(yx, 1+vec1) # s-by-d matrix
        g = -numpy.mean(vec2, axis=0).reshape(self.d, 1)+(self.lam / 2*numpy.sum(self.w * self.w)) # d-by-1 matrix
        return g
    
    #Compute q gradient 
    def gradient_local(self):
        for i in range(self.q):
            gradient = self.gradient()
            self.v = self.beta*self.v + gradient
            self.w -= self.alpha * self.v
            losses = self.loss()
            #print("local gradient ", gradient)
            print("The local objective function value is: ", losses)
        return self.w
    
    #Compute Stochastic gradient descent
    def stochgradient_descent(self, i, x_rand, y_rand):
        import numpy as np
        xi = x_rand[i,:]
        yi = float(y_rand[i,:])
        # calculate the stochastic gradient and the loss of local worker
        d = xi.shape[0]
        yx = yi * xi # 1-by-d matrix
        yxw = float(np.dot(yx, self.w)) #scalar
        loss = np.log(1 + np.exp(-yxw)) #scalar
        reg = lam / 2 * np.sum(self.w * self.w)#scalar
        obj = loss + reg
        g_loss = -yx.T / (1 + np.exp(yxw))# d-by-1 matrix
        g = g_loss + lam * self.w # d-by-1 matrix
        return g, obj
    
    #Compute q stochastic gradients 
    def stochgradient_descent_local(self):
        import numpy as np
        for i in range(self.q):
            rand_indices = np.random.permutation(self.s)
            x_rand = self.x[rand_indices,:]
            y_rand = self.y[rand_indices,:]
            objval = 0
            for i in range(self.s):
                gradient, obj = self.stochgradient_descent(i, x_rand, y_rand)
                self.w -= self.alpha * gradient
                objval+=obj
            self.alpha *= 0.9 # decrease step size
            objval /= self.s
            print("The local objective function value is given by: ", objval)
        return self.w
    
    #Find direction to be sent back to server
    def observe_direction(self):
        pk = self.initial_w - self.gradient_local()
        return pk
    
    #Finding direction for stochastic gradient to be sent back to server 
    def observe_stochdirection(self):
        pk = self.initial_w - self.stochgradient_descent_local()
        return pk

class Server:
    def __init__(self, m, n, d, alpha, lam):
        self.m = m #number of worker nodes
        self.n = n #number of training samples
        self.d = d #number of features
        self.w = numpy.zeros((d, 1)) #d-by-1 model parameter vector
        self.pk = numpy.zeros((d, 1)) #directions recieved from different workers
        self.alpha = alpha #Server Learning rate
        self.lam = lam #Server reg parameter
        self.loss = 0 #loss function value

    
    def broadcast(self):
        return self.w
    
    def aggregate(self, pk, losses):
        self.pk = numpy.mean(numpy.array(pk), axis=0)
        self.loss = sum(losses)
        
    def update(self):
        self.w = self.w - self.alpha*self.pk
        self.alpha *= 0.9 # decrease step size
            
    def objective_fun(self):
        reg = self.lam/2 * numpy.sum(self.w*self.w)
        self.obj = self.loss/self.m + reg
        return self.obj

def create_server_workers(m, x, y, q, alpha, lam, beta):
    n, d = x.shape
    s = math.floor(n / m)
